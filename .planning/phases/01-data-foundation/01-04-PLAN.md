    ```css
    /* ============================================
       Print Styles (D-14: browser print optimization)
       ============================================ */

    @media print {
      /* Hide all interactive/navigation UI elements */
      .no-print,
      nav,
      button,
      [role="tablist"] {
        display: none !important;
      }

      /* Reset page for print */
      body {
        background: white !important;
        color: black !important;
        font-size: 10pt;
        line-height: 1.4;
      }

      /* Page margins and size */
      @page {
        margin: 1cm;
        size: letter;
      }

      /* Headings for print */
      h1 {
        font-size: 18pt !important;
        font-weight: 600;
        line-height: 1.1;
      }

      h2 {
        font-size: 14pt !important;
        font-weight: 600;
        line-height: 1.2;
      }

      /* Prevent row groups from breaking across pages */
      .directory-group {
        break-inside: avoid;
      }

      /* Ensure table headers repeat on each page */
      thead {
        display: table-header-group;
      }

      /* Orphans/widows control */
      p, li {
        orphans: 3;
        widows: 3;
      }

      /* Alphabetical directory: two columns on print (D-15) */
      .alpha-directory {
        column-count: 2;
        column-gap: 1.5cm;
      }

      /* Show all directory views when printing (not just active tab) */
      .print-show {
        display: block !important;
      }

      /* Remove shadows, borders, backgrounds for clean print */
      .card, [class*="shadow"] {
        box-shadow: none !important;
        border: none !important;
      }

      /* Links: show as plain text */
      a {
        color: black !important;
        text-decoration: none !important;
      }

      /* Add spacing between directory sections in print */
      .print-section-break {
        margin-top: 1cm;
        page-break-before: auto;
      }
    }
    ```
